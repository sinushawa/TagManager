167 public static IEnumerable<T> FindChildren<T>(this DependencyObject parent, Func<T, bool> predicate)
168 where T : DependencyObject
169 {
170 var children = new List<DependencyObject>();
171
172 if ((parent is Visual) || (parent is Visual3D))
173 {
174 var visualChildrenCount = VisualTreeHelper.GetChildrenCount(parent);
175 for (int childIndex = 0; childIndex < visualChildrenCount; childIndex++)
176 children.Add(VisualTreeHelper.GetChild(parent, childIndex));
177 }
178 foreach (var logicalChild in LogicalTreeHelper.GetChildren(parent).OfType<DependencyObject>())
179 if (!children.Contains(logicalChild))
180 children.Add(logicalChild);
181
182 foreach (var child in children)
183 {
184 var typedChild = child as T;
185 if ((typedChild != null) && predicate.Invoke(typedChild))
186 yield return typedChild;
187
188 foreach (var foundDescendant in FindChildren(child, predicate))
189 yield return foundDescendant;
190 }
191 yield break;
192 }
